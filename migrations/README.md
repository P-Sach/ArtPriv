"""
Alembic migration script for initial schema
Run after setting up Alembic with: alembic init alembic
"""

# To use this with Alembic:
# 1. Install alembic: pip install alembic
# 2. Initialize: alembic init alembic
# 3. Update alembic.ini with your DATABASE_URL
# 4. Update alembic/env.py to import your Base from database.py
# 5. Generate migration: alembic revision --autogenerate -m "Initial schema"
# 6. Apply migration: alembic upgrade head
